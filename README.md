# Usage
from a2 import aich

search_word = input('Enter your encryption phrase:')

encrypted = aich.aichin(search_word)

print (encrypted)

search_word = input('Enter your decryption phrase:')

encrypted = aich.aichout(search_word)

print (encrypted)
